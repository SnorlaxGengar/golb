Keyed_caesar

20 April 2021

基于密钥的凯撒密码

密文：s0a6u3u1s0bv1a 密钥：guangtou 偏移：6,20,0,13,6,19,14,20 明文：y0u6u3h1y0uj1u

代码

#include "catch2/catch_all.hpp"
#include <string>
#include <iostream>

TEST_CASE("Keyed Caesar", "[Keyed Caesar]") {
    std::string cipherText("s0a6u3u1s0bv1a");
    std::string key("guangtou");
    auto it = key.begin();
    std::transform(cipherText.begin(), cipherText.end(), cipherText.begin(), [&it, &key](char c) {
        if (!std::isalpha(c)) {
            return c;
        }

        char k = *(it++);

        if (it == key.end()) {
            it = key.begin();
        }

        // 需要处理偏移后溢出'z'的情况
        return static_cast<char>((c - 'a' + k - 'a') % 26 + 'a');
    });
    REQUIRE(cipherText == "y0u6u3h1y0uj1u");
}
